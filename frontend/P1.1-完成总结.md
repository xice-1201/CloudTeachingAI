```
  /learning/path          # 学习路线
  /assignments            # 作业列表
  /assignments/:id        # 作业详情
  /assignments/:id/submissions  # 批改列表（教师/管理员）
  /chat                   # AI 助手
  /notifications          # 通知中心
  /profile                # 个人资料
  /mentor                 # 导师关系
  /admin                  # 系统管理（管理员）
```
